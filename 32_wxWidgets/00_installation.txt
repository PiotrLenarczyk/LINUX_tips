UBUNTU INSTALLATION:
installing wxWidgets https://wiki.wxwidgets.org/Installing_and_configuring_under_Ubuntu
get repository 3.0 https://wiki.codelite.org/pmwiki.php/Main/WxWidgets30Binaries	
 sudo apt-key adv --fetch-keys http://repos.codelite.org/CodeLite.asc
 sudo apt-add-repository 'deb http://repos.codelite.org/wx3.0.4/ubuntu/ artful universe'
 sudo apt-get update 
 sudo apt-get install libwxbase3.0-0-unofficial \
                 libwxbase3.0-dev \
                 libwxgtk3.0-0-unofficial \
                 libwxgtk3.0-dev \
                 wx3.0-headers \
                 wx-common \
                 libwxbase3.0-dbg \
                 libwxgtk3.0-dbg \
                 wx3.0-i18n \
                 wx3.0-examples \
                 wx3.0-doc
